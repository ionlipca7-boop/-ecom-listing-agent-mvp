import csv
import json
from pathlib import Path

BASE_DIR = Path(__file__).resolve().parent
UPLOAD_RESULTS_DIR = BASE_DIR / "storage" / "upload_results"
OUTPUT_JSON = BASE_DIR / "storage" / "exports" / "ebay_upload_result_parsed_v1.json"

SUCCESS_WORDS = ["success", "successful", "uploaded", "created", "added", "ok"]
ERROR_WORDS = ["error", "failed", "invalid", "missing", "blocked", "not accepted"]
WARNING_WORDS = ["warning", "warn", "review", "check"]

def normalize_text(value):
    if value is None:
        return ""
    return str(value).strip()

def detect_status(text):
    lowered = normalize_text(text).lower()
    if any(word in lowered for word in ERROR_WORDS):
        return "ERROR"
    if any(word in lowered for word in WARNING_WORDS):
        return "WARNING"
    if any(word in lowered for word in SUCCESS_WORDS):
        return "SUCCESS"
    return "UNKNOWN"

def parse_csv_file(path: Path):
    rows = []
    with path.open("r", encoding="utf-8-sig", newline="") as f:
        reader = csv.DictReader(f, delimiter=";")
        if reader.fieldnames is None:
            return []
        for idx, row in enumerate(reader, start=1):
            combined = " | ".join([normalize_text(v) for v in row.values() if normalize_text(v)])
            rows.append({
                "row_index_1_based": idx,
                "status": detect_status(combined),
                "raw": row,
                "combined_text": combined,
            })
    return rows

def parse_text_file(path: Path):
    lines = []
    with path.open("r", encoding="utf-8-sig", errors="replace") as f:
        for idx, line in enumerate(f, start=1):
            text = normalize_text(line)
            if not text:
                continue
            lines.append({
                "line_index_1_based": idx,
                "status": detect_status(text),
                "text": text,
            })
    return lines

def choose_latest_file():
    candidates = []
    for pattern in ("*.csv", "*.txt", "*.log"):
        candidates.extend(UPLOAD_RESULTS_DIR.glob(pattern))
    if not candidates:
        return None
    candidates.sort(key=lambda p: p.stat().st_mtime, reverse=True)
    return candidates[0]

def main():
    UPLOAD_RESULTS_DIR.mkdir(parents=True, exist_ok=True)
    latest_file = choose_latest_file()
    if latest_file is None:
        output = {
            "parser_status": "NO_INPUT",
            "upload_results_dir": str(UPLOAD_RESULTS_DIR),
            "message": "Place eBay upload result file into storage/upload_results and run again",
            "latest_file": "",
            "summary": {"success_count": 0, "warning_count": 0, "error_count": 0, "unknown_count": 0},
            "items": [],
        }
    else:
        suffix = latest_file.suffix.lower()
        if suffix == ".csv":
            items = parse_csv_file(latest_file)
        else:
            items = parse_text_file(latest_file)
        success_count = sum(1 for item in items if item.get("status") == "SUCCESS")
        warning_count = sum(1 for item in items if item.get("status") == "WARNING")
        error_count = sum(1 for item in items if item.get("status") == "ERROR")
        unknown_count = sum(1 for item in items if item.get("status") == "UNKNOWN")
        overall = "SUCCESS" if error_count == 0 and len(items) > 0 else "HAS_ERRORS"
        output = {
            "parser_status": overall,
            "upload_results_dir": str(UPLOAD_RESULTS_DIR),
            "latest_file": str(latest_file),
            "summary": {
                "success_count": success_count,
                "warning_count": warning_count,
                "error_count": error_count,
                "unknown_count": unknown_count,
            },
            "items": items,
        }
    OUTPUT_JSON.parent.mkdir(parents=True, exist_ok=True)
    with OUTPUT_JSON.open("w", encoding="utf-8") as f:
        json.dump(output, f, ensure_ascii=False, indent=2)
    print("EBAY_UPLOAD_RESULT_PARSER_V1:")
    print(f"parser_status: {output['parser_status']}")
    print(f"latest_file: {output['latest_file']}")
    print(f"success_count: {output['summary']['success_count']}")
    print(f"warning_count: {output['summary']['warning_count']}")
    print(f"error_count: {output['summary']['error_count']}")
    print(f"unknown_count: {output['summary']['unknown_count']}")
    print(f"output_json: {OUTPUT_JSON}")

if __name__ == "__main__":
    main()
